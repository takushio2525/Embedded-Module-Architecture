% API仕様
\section{API仕様}

\subsection{IModuleインターフェース}

\begin{apibox}{IModule — モジュールインターフェース}
\textbf{定義ファイル:} \texttt{lib/ModuleCore/IModule.h}

全ハードウェアモジュールが実装すべき抽象インターフェース。
\texttt{SystemData}の前方宣言のみを使用し、プロジェクト固有のヘッダーに依存しない。
\end{apibox}

\begin{lstlisting}[style=cppstyle, caption={IModuleインターフェース定義}]
// IModule.h — モジュールインターフェース（プロジェクト非依存）
#pragma once

// SystemDataの前方宣言
// 各プロジェクトでSystemData構造体を定義すること
struct SystemData;

class IModule {
public:
    // 仮想デストラクタ
    virtual ~IModule() {}

    // モジュールの初期化
    // 戻り値: true=成功, false=失敗
    virtual bool init() = 0;

    // 入力フェーズ更新（センサー読み取り等）
    // 入力を持つモジュールのみオーバーライドする
    virtual void updateInput(SystemData& data) {}

    // 出力フェーズ更新（アクチュエータ制御・画面描画等）
    // 出力を持つモジュールのみオーバーライドする
    virtual void updateOutput(SystemData& data) {}

    // モジュールの終了処理（リソース解放）
    // デフォルトは空実装。解放が必要なモジュールのみオーバーライド
    virtual void deinit() {}

    // 動的有効/無効化フラグ
    // false の場合、ループ内で updateInput()/updateOutput() がスキップされる
    bool enabled = true;
};
\end{lstlisting}

\begin{notebox}[前方宣言によるプロジェクト非依存性]
\texttt{IModule.h}は\texttt{SystemData}を前方宣言のみで参照しているため、
\texttt{SystemData.h}を\texttt{\#include}する必要がない。\\
\texttt{updateInput()}/\texttt{updateOutput()}の引数は参照型（\texttt{SystemData\&}）であるため、前方宣言だけで十分であり、
どのプロジェクトにもそのままコピーして流用できる。
\end{notebox}

\begin{cautionbox}[データアクセスルール]
\texttt{updateInput()}/\texttt{updateOutput()}内では、\textbf{自モジュールのサブ構造体のみ}にアクセスすること。
他モジュールのサブ構造体への読み書きは禁止とする。
モジュール間のデータ連携が必要な場合は、必ずロジックフェーズを経由する。
\end{cautionbox}

\begin{notebox}[\texttt{SystemData}名の標準化]
\texttt{IModule.h}は前方宣言で\texttt{SystemData}という型名に依存している。
この名前は本アーキテクチャの標準命名規則として固定されており、
すべてのプロジェクトで\texttt{SystemData}という名前の構造体を\texttt{include/SystemData.h}に定義する。
\end{notebox}

\subsubsection{使用例}

\begin{lstlisting}[style=cppstyle, caption={出力モジュールの例（ヘッダー）}]
// LedModule.h
#pragma once
#include <Arduino.h>
#include "IModule.h"

// モジュール固有のConfig構造体
struct LedConfig {
    uint8_t ledPin;  // LED出力ピン
};

// モジュール固有のデータ構造体（メンバにデフォルト値を明示）
struct LedData {
    bool state = false;
    uint8_t brightness = 0;
};

// SystemDataの前方宣言はIModule.h内で行われている

// IModuleを継承（出力専用のためupdateOutput()のみオーバーライド）
class LedModule : public IModule {
private:
    LedConfig _config;  // Configをメンバとして保持
public:
    // コンストラクタでConfig構造体を受け取る
    LedModule(const LedConfig& config);
    bool init() override;
    void updateOutput(SystemData& data) override;
};
\end{lstlisting}

\begin{lstlisting}[style=cppstyle, caption={出力モジュールの例（実装ファイル）}]
// LedModule.cpp
#include "LedModule.h"
#include "SystemData.h"  // SystemDataの完全な定義を取得

LedModule::LedModule(const LedConfig& config) : _config(config) {}

bool LedModule::init() {
    pinMode(_config.ledPin, OUTPUT);  // Configのピンを使用
    Serial.println("[Led] init OK");
    return true;
}

// data はポインタではなく参照なので、ドット演算子でアクセス
void LedModule::updateOutput(SystemData& data) {
    digitalWrite(_config.ledPin, data.led.state);
}
\end{lstlisting}

\subsection{ModuleTimerクラス}

\begin{apibox}{ModuleTimer — ノンブロッキングタイマー}
\textbf{定義ファイル:} \texttt{lib/ModuleCore/ModuleTimer.h}

\texttt{millis()}ベースのシンプルなタイマークラス。
\texttt{delay()}を使用せずに周期実行・デバウンス・ワンショット・タイムアウト等を実現する。
インスタンスを複数作成することで、モジュール内で複数のタイマーを独立に管理できる。
IModuleと同様にプロジェクト非依存であり、そのまま流用可能。
\end{apibox}

\begin{lstlisting}[style=cppstyle, caption={ModuleTimerクラス定義}]
// ModuleTimer.h — プロジェクト非依存のコアヘッダー
#pragma once
#include <Arduino.h>

class ModuleTimer {
private:
    unsigned long _startTime;

public:
    // コンストラクタ
    ModuleTimer() : _startTime(0) {}

    // 基準時刻をセット
    // 引数なし: getNowTime()が0から開始
    // 引数あり: getNowTime()が指定値から開始
    void setTime(unsigned long offsetMs = 0) {
        _startTime = millis() - offsetMs;
    }

    // 基準時刻からの経過時間(ms)を取得
    unsigned long getNowTime() const {
        return millis() - _startTime;
    }
};
\end{lstlisting}

\subsubsection{setTime(unsigned long offsetMs = 0)}

\begin{tabularx}{\textwidth}{l X}
\toprule
\textbf{項目} & \textbf{内容} \\
\midrule
引数 & \texttt{offsetMs}（デフォルト: 0） — 初期オフセット値(ms) \\
戻り値 & \texttt{void} \\
動作 & 内部の基準時刻を\texttt{millis() - offsetMs}に設定する。
       \texttt{offsetMs = 0}の場合、直後の\texttt{getNowTime()}は0を返す。
       \texttt{offsetMs = 100}の場合、直後の\texttt{getNowTime()}は100を返す \\
用途 & タイマーの開始・リセット。\texttt{init()}内および周期リセット時に使用 \\
\bottomrule
\end{tabularx}

\subsubsection{getNowTime() const}

\begin{tabularx}{\textwidth}{l X}
\toprule
\textbf{項目} & \textbf{内容} \\
\midrule
引数 & なし \\
戻り値 & \texttt{unsigned long} — 最後の\texttt{setTime()}からの経過時間(ms) \\
動作 & \texttt{millis() - \_startTime}を返す \\
用途 & 経過時間の判定。条件分岐と組み合わせて周期実行・デバウンス等を実現 \\
\bottomrule
\end{tabularx}

\subsubsection{使用パターン}

\begin{lstlisting}[style=cppstyle, caption={周期実行パターン}]
ModuleTimer readTimer;

bool init() override {
    readTimer.setTime();
    return true;
}

void updateInput(SystemData& data) override {
    if (readTimer.getNowTime() >= 1000) {
        readTimer.setTime();  // リセット
        data.temp.temperature = readSensor();
    }
}
\end{lstlisting}

\begin{lstlisting}[style=cppstyle, caption={デバウンスパターン}]
ModuleTimer debounce;

void updateInput(SystemData& data) override {
    bool rawInput = digitalRead(BUTTON_PIN);

    if (rawInput != lastRawInput) {
        debounce.setTime();  // 入力変化でリセット
        lastRawInput = rawInput;
    }

    if (debounce.getNowTime() >= 50) {
        data.button.pressed = lastRawInput;  // 50ms安定で確定
    }
}
\end{lstlisting}

\subsection{ProjectConfigファイルとSystemData}

\begin{apibox}{ProjectConfig — Configインスタンスの集約}
\textbf{定義ファイル:} \texttt{include/ProjectConfig.h}（プロジェクト固有）

各モジュールのConfigインスタンス（ピンアサイン等）を集約した、プロジェクト固有の設定ファイル。
\texttt{main.cpp}のみがインクルードする。
\end{apibox}

\begin{apibox}{SystemData — モジュール間共有データ構造体}
\textbf{定義ファイル:} \texttt{include/SystemData.h}（プロジェクト固有）

各モジュールの\texttt{Data}構造体を集約した共有データ構造体。
\texttt{IModule}の\texttt{updateInput()}/\texttt{updateOutput()}メソッドの引数型として使用される。
モジュールの\texttt{.cpp}ファイルからインクルードされる。
\end{apibox}

\subsubsection{設計ルール}

\begin{enumerate}
    \item 各モジュールは、自身の\texttt{Config}構造体（型定義）と\texttt{Data}構造体（型定義）をモジュールのヘッダーファイル内に定義する
    \item \texttt{ProjectConfig.h}には、Configの\textbf{インスタンス}（実際のピン番号等の値）のみを定義する
    \item モジュール固有のピン番号は\texttt{constexpr}単体定数にせず、Configインスタンスのリテラルに\textbf{直書き}する。外部からピンを参照する場合は\texttt{SERVO\_CONFIG.pin}のようにConfig構造体経由でアクセスする
    \item 共有バスピン（\texttt{SPI\_MOSI\_PIN}等）は特定モジュールに属さないため、\texttt{constexpr}単体定数として定義してよい
    \item \texttt{SystemData}構造体は\texttt{SystemData.h}に分離し、プロジェクトで使用するモジュールのData構造体をメンバとして持つ
    \item モジュールの\texttt{.cpp}は\texttt{SystemData.h}をインクルードする（\texttt{ProjectConfig.h}はインクルードしない）
    \item モジュールは\texttt{updateInput()}/\texttt{updateOutput()}内で、\textbf{自身のData構造体のみ}にアクセスする（他モジュールのData構造体への読み書きは禁止）
\end{enumerate}

\begin{lstlisting}[style=cppstyle, caption={SystemData.h構成例}]
// SystemData.h — モジュール間共有データ構造体
#pragma once
#include "LedModule.h"
#include "TempSensorModule.h"
#include "CameraModule.h"

struct SystemData {
    LedData led;
    TempData temp;
    CameraData camera;
};
\end{lstlisting}

\begin{lstlisting}[style=cppstyle, caption={ProjectConfig.h構成例}]
// ProjectConfig.h — 共有バスピン + Configインスタンス
#pragma once
#include "SystemData.h"

// ===== 共有バスピン（単体定数OK） =====
constexpr int SPI_MOSI_PIN = 38;
constexpr int SPI_SCK_PIN  = 39;
constexpr int I2C_SDA_PIN  = 2;
constexpr int I2C_SCL_PIN  = 3;

// ===== Configインスタンス（ピンは直書き） =====
const LedConfig LED_CONFIG = { .ledPin = 13 };
const TempSensorConfig TEMP_CONFIG = { .tempPin = 36 };
const CameraConfig CAMERA_CONFIG = { .sdaPin = 21, .sclPin = 22 };
\end{lstlisting}

\begin{notebox}[ファイル分離の理由]
\texttt{SystemData.h}を\texttt{ProjectConfig.h}から分離することで、
モジュールの\texttt{.cpp}ファイルは\texttt{ProjectConfig.h}を知る必要がなくなる。\\
\textbf{SystemData.h} — 各モジュールのData構造体を集約。モジュールの\texttt{.cpp}がインクルード。\\
\textbf{ProjectConfig.h} — Configインスタンス（ピン番号等の具体的な値）のみ。
\texttt{main.cpp}だけがインクルードすればよい。
回路設計担当者はこのファイルのConfig定義だけを参照すれば回路設計が可能。
\end{notebox}

\subsection{モジュールConfig構造体}

\begin{apibox}{モジュールConfig — ハードウェア設定構造体}
各モジュールが必要とするピンアサインや設定をまとめた構造体。
各モジュールのヘッダーファイル内で型を定義し、
\texttt{ProjectConfig.h}内で具体的な値を持つインスタンスを定義する。
\end{apibox}

\subsubsection{命名規則}

\begin{tabularx}{\textwidth}{l l X}
\toprule
\textbf{要素} & \textbf{命名規則} & \textbf{例} \\
\midrule
Config構造体の型名 & \texttt{\{Module\}Config} & \texttt{LedConfig}, \texttt{CameraConfig} \\
Configインスタンス名 & \texttt{\{MODULE\}\_CONFIG}(大文字) & \texttt{LED\_CONFIG}, \texttt{CAMERA\_CONFIG} \\
Data構造体の型名 & \texttt{\{Module\}Data} & \texttt{LedData}, \texttt{CameraData} \\
\bottomrule
\end{tabularx}

\subsubsection{設計ルール}

\begin{enumerate}
    \item Config構造体の\textbf{型定義}は、各モジュールのヘッダーファイル内に記述する
    \item Config構造体の\textbf{インスタンス}（実際のピン番号等）は\texttt{ProjectConfig.h}に集約する
    \item モジュールのコンストラクタは\texttt{const \{Module\}Config\&}（const参照）でConfigを受け取る
    \item 受け取ったConfigは、メンバ変数\texttt{\_config}としてコピー保持する（Configは初期値として扱う）
    \item 実行時に設定値の変更が必要な場合は、\texttt{\_config}メンバを直接変更するか、セッターメソッドを用意する
    \item ピン番号はハードコードせず、必ず\texttt{\_config}経由でアクセスする
    \item 通信バスを使用するモジュールは、バスポインタをConfig構造体に含めず、コンストラクタの引数で受け取る（§通信バスパターン参照）
\end{enumerate}

\begin{lstlisting}[style=cppstyle, caption={Config構造体の定義と使用例}]
// TempSensorModule.h
#pragma once
#include "IModule.h"
#include "ModuleTimer.h"

// --- Config構造体（ピンアサイン・設定） ---
struct TempSensorConfig {
    uint8_t tempPin;          // アナログ入力ピン
    unsigned long interval;   // 読み取り間隔(ms)
};

// --- Data構造体（メンバにデフォルト値を明示） ---
struct TempData {
    float temperature = -999.0f;
    bool isValid = false;
};

class TempSensorModule : public IModule {
private:
    TempSensorConfig _config;
    ModuleTimer _readTimer;
public:
    // コンストラクタ: Config構造体をconst参照で受け取る
    TempSensorModule(const TempSensorConfig& config)
        : _config(config) {}

    bool init() override {
        pinMode(_config.tempPin, INPUT);
        _readTimer.setTime();
        return true;
    }

    void updateInput(SystemData& data) override {
        if (_readTimer.getNowTime() >= _config.interval) {
            _readTimer.setTime();
            data.temp.temperature = analogRead(_config.tempPin);
            data.temp.isValid = true;
        }
    }
};
\end{lstlisting}

\begin{lstlisting}[style=cppstyle, caption={通信バスを使用するモジュールのConfig構造体とコンストラクタ例}]
// SpiSensorModule.h — Config構造体にはバスポインタを含めない
struct SpiSensorConfig {
    uint8_t csPin;            // チップセレクトピン
    uint32_t clockSpeed;      // SPIクロック速度(Hz)
    unsigned long interval;   // 読み取り間隔(ms)
};

class SpiSensorModule : public IModule {
private:
    SpiSensorConfig _config;
    SPIClass* _spi;  // バスポインタはメンバ変数として保持
public:
    // Config + バスポインタをコンストラクタで受け取る
    SpiSensorModule(const SpiSensorConfig& config, SPIClass* spi)
        : _config(config), _spi(spi) {}
};

// ProjectConfig.h — Configは純粋な設定値のみ
const SpiSensorConfig SPI_SENSOR_CONFIG = {
    .csPin = 5, .clockSpeed = 1000000, .interval = 100
};

// main.cpp — バスの紐付けはモジュール生成時に行う
SPIClass spi(HSPI);
SpiSensorModule spiModule(SPI_SENSOR_CONFIG, &spi);
\end{lstlisting}
