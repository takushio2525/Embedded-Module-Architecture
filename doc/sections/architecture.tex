% システムアーキテクチャ
\section{システムアーキテクチャ}

\subsection{全体構造}

本アーキテクチャは、以下のコア要素で構成される：

\begin{enumerate}
    \item \textbf{IModule<T>}（テンプレートインターフェース） — モジュールのライフサイクルを定義。プロジェクト非依存
    \item \textbf{ModuleTimer}（ユーティリティクラス） — ノンブロッキングなタイマー制御。プロジェクト非依存
    \item \textbf{SystemData}（共有データ構造体） — モジュール間のデータ連携。プロジェクト固有
    \item \textbf{モジュールConfig}（設定構造体） — 各モジュールのピンアサインや設定を構造体で定義。プロジェクト固有
\end{enumerate}

\begin{notebox}[コア層とプロジェクト層の分離]
\texttt{IModule<T>}と\texttt{ModuleTimer}は\textbf{コア層}に属し、プロジェクト固有の型を一切知らない。\\
\texttt{SystemData}とモジュールConfigは\textbf{プロジェクト層}に属し、
前者は使用するモジュールのサブ構造体を集約し、後者はハードウェア構成（ピンアサイン等）を定義する。\\
テンプレートパラメータ\texttt{T}を用いることで、コア層の完全な独立性を実現している。
\end{notebox}

\subsection{実行フロー}

プログラムの実行フローは、ゲームループパターンに基づき、3フェーズモデルで構成される：

\begin{lstlisting}[style=cppstyle, caption={3フェーズモデルによる実行フロー}]
// main.cpp
#include "IModule.h"
#include "ProjectConfig.h"  // SystemData + Config定義
#include "TempSensorModule.h"
#include "SwitchModule.h"
#include "LedModule.h"
#include "MotorModule.h"

// システムデータ
SystemData systemData;

// 入力モジュール（Config構造体を渡して生成）
TempSensorModule tempModule(TEMP_CONFIG);
SwitchModule swModule(SW_CONFIG);

// 出力モジュール
LedModule ledModule(LED_CONFIG);
MotorModule motorModule(MOTOR_CONFIG);

// 入力モジュール配列
IModule<SystemData>* inputModules[] = {
    &tempModule,
    &swModule,
};
const int INPUT_COUNT = sizeof(inputModules) / sizeof(inputModules[0]);

// 出力モジュール配列
IModule<SystemData>* outputModules[] = {
    &ledModule,
    &motorModule,
};
const int OUTPUT_COUNT = sizeof(outputModules) / sizeof(outputModules[0]);

// ロジックフェーズ（標準は関数として実装）
void applyPattern(SystemData& data) {
    // 入力SystemDataを読み、出力SystemDataを書き換える
    data.led.state = data.sw.isPressed;
    data.motor.speed = (data.temp.temperature > 30.0) ? 255 : 0;
}

void setup() {
    Serial.begin(115200);

    // 全入力モジュールを初期化
    for (int i = 0; i < INPUT_COUNT; i++) {
        if (!inputModules[i]->init()) {
            Serial.printf("Input Module %d: init failed\n", i);
        }
    }
    // 全出力モジュールを初期化
    for (int i = 0; i < OUTPUT_COUNT; i++) {
        if (!outputModules[i]->init()) {
            Serial.printf("Output Module %d: init failed\n", i);
        }
    }
}

void loop() {
    // 1. 入力フェーズ：センサー値をSystemDataに書き込む
    for (int i = 0; i < INPUT_COUNT; i++) {
        inputModules[i]->update(systemData);
    }
    // 2. ロジックフェーズ：入力→出力のSystemData変換
    applyPattern(systemData);
    // 3. 出力フェーズ：SystemDataの値に従いハードウェア制御
    for (int i = 0; i < OUTPUT_COUNT; i++) {
        outputModules[i]->update(systemData);
    }
}
\end{lstlisting}

\subsection{設計パターン}

\subsubsection{モジュール配列パターン}
\texttt{inputModules[]}と\texttt{outputModules[]}の2つの配列でモジュールを管理する。
配列の順序がそのまま各フェーズ内の実行順序となる。
これにより：
\begin{itemize}
    \item 入力と出力の処理順序が明示的かつ決定論的
    \item 入力→ロジック→出力のデータの流れが保証される
    \item 1ループにつき各モジュール1回の更新が保証される
    \item 各モジュールが独立した「道具」として機能する
\end{itemize}

\begin{notebox}[重要：処理順序]
各フェーズ内での配列順序はデータフローの順序を意味する。\\
例：温度センサーの値を補正センサーで校正する場合、温度センサーを補正センサーより前に配置する。\\
ただし、入力と出力のクロス参照（出力モジュールが入力のSystemDataを読む等）は行わない。
\end{notebox}

\subsubsection{SystemDataパターン}
各モジュールは固有のサブ構造体を定義し、それらをプロジェクト固有の\texttt{SystemData}構造体に集約する。

\begin{lstlisting}[style=cppstyle, caption={SystemDataパターンの構造}]
// 各モジュールが自身のデータ構造体を定義
// LedModule.h 内
struct LedData {
    bool state;
    uint8_t brightness;
};

// TempSensorModule.h 内
struct TempData {
    float temperature;
    bool isValid;
    bool hasError;
};

// プロジェクト固有: SystemData.h
struct SystemData {
    LedData led;
    TempData temp;
    // 使用するモジュールの構造体を追加するだけ
};
\end{lstlisting}

\subsubsection{モジュールConfigパターン}
各モジュールは、自身に必要なピンアサインや設定を\texttt{Config}構造体として定義する。
モジュールのコンストラクタではピン番号を個別に受け取るのではなく、Config構造体を受け取ることで、
ハードウェア構成を明示的かつ安全に管理する。

\begin{funcblock}{Configパターンの利点}
\begin{itemize}
    \item コンストラクタの引数順序を覚える必要がない（構造体のメンバ名で明示的に指定）
    \item 全モジュールのピンアサインが\texttt{ProjectConfig.h}に集約され、一覧性が高い
    \item 回路設計担当者は\texttt{ProjectConfig.h}のConfig定義セクションだけを見れば回路設計が可能
    \item 将来的にピン以外の設定（I2Cアドレス、動作モード等）もConfig構造体に追加して拡張可能
\end{itemize}
\end{funcblock}

\begin{lstlisting}[style=cppstyle, caption={モジュールConfigパターンの例}]
// LedModule.h 内 — Config構造体を定義
struct LedConfig {
    uint8_t ledPin;      // LED出力ピン
};

// モジュールクラスはコンストラクタでConfigを受け取る
class LedModule : public IModule<SystemData> {
private:
    LedConfig _config;
public:
    LedModule(const LedConfig& config) : _config(config) {}
    // ...
};

// ProjectConfig.h 内 — プロジェクトのピンアサインを集約
const LedConfig LED_CONFIG = { .ledPin = 13 };

// main.cpp — Config定数を渡してインスタンス生成
LedModule ledModule(LED_CONFIG);
\end{lstlisting}

\subsubsection{通信バスパターン}
SPI・I2C等の通信バスは複数モジュールで共有されるリソースであり、
個々のモジュールが所有・初期化すべきではない。
バスインスタンスを\texttt{main.cpp}で生成し、Config構造体のポインタ経由でモジュールに渡す。

\begin{funcblock}{通信バスを共有リソースとして管理する理由}
\begin{itemize}
    \item 複数モジュールが同じバスを使う場合、各モジュールが\texttt{begin()}すると競合・二重初期化が発生する
    \item バスのピンアサインやインスタンスがモジュール内にハードコードされると、モジュールの再利用性が損なわれる
    \item バスの生成と初期化を\texttt{main.cpp}に集約することで、共有リソースの管理が明確になる
\end{itemize}
\end{funcblock}

\paragraph{SPIバスの例}

\begin{lstlisting}[style=cppstyle, caption={SPIバスを使用するモジュールのConfig構造体と実装例}]
// SpiSensorModule.h
#pragma once
#include <SPI.h>
#include "IModule.h"
#include "ModuleTimer.h"

// --- Config構造体（バスポインタ + デバイス固有設定） ---
struct SpiSensorConfig {
    SPIClass* spi;            // SPIバスへのポインタ
    uint8_t csPin;            // チップセレクトピン
    uint32_t clockSpeed;      // SPIクロック速度(Hz)
    unsigned long interval;   // 読み取り間隔(ms)
};

// --- Data構造体 ---
struct SpiSensorData {
    uint16_t rawValue;
    bool hasError;
};

class SpiSensorModule : public IModule<SystemData> {
private:
    SpiSensorConfig _config;
    ModuleTimer _readTimer;
public:
    SpiSensorModule(const SpiSensorConfig& config)
        : _config(config) {}

    bool init() override {
        // CSピンの設定のみ行う（bus.begin()は呼ばない）
        pinMode(_config.csPin, OUTPUT);
        digitalWrite(_config.csPin, HIGH);
        _readTimer.setTime();
        return true;
    }

    void update(SystemData& data) override {
        if (_readTimer.getNowTime() >= _config.interval) {
            _readTimer.setTime();
            // beginTransaction/endTransactionで排他制御
            _config.spi->beginTransaction(
                SPISettings(_config.clockSpeed, MSBFIRST, SPI_MODE0)
            );
            digitalWrite(_config.csPin, LOW);
            data.spiSensor.rawValue =
                _config.spi->transfer16(0x00);
            digitalWrite(_config.csPin, HIGH);
            _config.spi->endTransaction();
        }
    }
};
\end{lstlisting}

\paragraph{I2Cバスの例}

\begin{lstlisting}[style=cppstyle, caption={I2Cバスを使用するモジュールのConfig構造体と実装例}]
// I2cSensorModule.h
#pragma once
#include <Wire.h>
#include "IModule.h"
#include "ModuleTimer.h"

// --- Config構造体（バスポインタ + デバイスアドレス） ---
struct I2cSensorConfig {
    TwoWire* wire;            // I2Cバスへのポインタ
    uint8_t address;          // I2Cデバイスアドレス
    unsigned long interval;   // 読み取り間隔(ms)
};

// --- Data構造体 ---
struct I2cSensorData {
    float value;
    bool hasError;
};

class I2cSensorModule : public IModule<SystemData> {
private:
    I2cSensorConfig _config;
    ModuleTimer _readTimer;
public:
    I2cSensorModule(const I2cSensorConfig& config)
        : _config(config) {}

    bool init() override {
        // バスの初期化は行わない（main.cppで実施済み）
        _readTimer.setTime();
        return true;
    }

    void update(SystemData& data) override {
        if (_readTimer.getNowTime() >= _config.interval) {
            _readTimer.setTime();
            _config.wire->beginTransmission(_config.address);
            _config.wire->write(0x00);  // レジスタアドレス
            if (_config.wire->endTransmission() != 0) {
                data.i2cSensor.hasError = true;
                return;
            }
            _config.wire->requestFrom(_config.address,
                                      (uint8_t)2);
            if (_config.wire->available() >= 2) {
                uint8_t high = _config.wire->read();
                uint8_t low = _config.wire->read();
                data.i2cSensor.value =
                    (float)((high << 8) | low) / 100.0;
                data.i2cSensor.hasError = false;
            }
        }
    }
};
\end{lstlisting}

\paragraph{main.cppでのバス生成・初期化}

\begin{lstlisting}[style=cppstyle, caption={main.cppでの通信バス生成と初期化}]
// main.cpp
#include <SPI.h>
#include <Wire.h>
#include "ProjectConfig.h"
#include "SpiSensorModule.h"
#include "I2cSensorModule.h"

// バスインスタンスの生成（グローバルスコープ）
SPIClass spi(HSPI);
TwoWire wire(0);

// Configにバスポインタを含めて定義（ProjectConfig.h内）
// const SpiSensorConfig SPI_SENSOR_CONFIG = {
//     .spi = &spi, .csPin = 5,
//     .clockSpeed = 1000000, .interval = 100
// };
// const I2cSensorConfig I2C_SENSOR_CONFIG = {
//     .wire = &wire, .address = 0x48, .interval = 500
// };

SpiSensorModule spiModule(SPI_SENSOR_CONFIG);
I2cSensorModule i2cModule(I2C_SENSOR_CONFIG);

void setup() {
    Serial.begin(115200);

    // バス初期化（全モジュールのinit()より前）
    spi.begin(SCK_PIN, MISO_PIN, MOSI_PIN);
    wire.begin(SDA_PIN, SCL_PIN);

    // モジュール初期化
    spiModule.init();
    i2cModule.init();
}
\end{lstlisting}

\begin{notebox}[通信バスパターンのルール]
\begin{enumerate}
    \item バスインスタンスは\texttt{main.cpp}のグローバルスコープで生成する
    \item \texttt{bus.begin()}は\texttt{setup()}内で、全モジュールの\texttt{init()}より前に呼ぶ
    \item Config構造体にバスポインタ（\texttt{SPIClass*}, \texttt{TwoWire*}）を含める
    \item モジュールの\texttt{init()}で\texttt{bus.begin()}を呼ばない（CSピン設定等のみ行う）
    \item SPI使用時は\texttt{beginTransaction()}/\texttt{endTransaction()}で排他制御する
\end{enumerate}
\end{notebox}

\paragraph{アンチパターン：モジュール内でバスを生成・初期化する}

\begin{lstlisting}[style=cppstyle, caption={アンチパターン：モジュール内でバスを生成（禁止）}]
// NG: モジュール内でバスインスタンスを生成・初期化している
class BadSpiModule : public IModule<SystemData> {
private:
    SPIClass _spi(HSPI);  // モジュールがバスを所有
public:
    bool init() override {
        _spi.begin(18, 19, 23);  // ピンがハードコード
        return true;
    }
    // 別のモジュールも同じバスをbegin()すると競合する
};
\end{lstlisting}

\subsubsection{エラーハンドリングパターン}
\begin{itemize}
    \item \texttt{init()} — 戻り値\texttt{bool}で初期化の成否を返す。失敗時はモジュールを無効化可能
    \item \texttt{update()} — エラー発生時は\texttt{SystemData}内のサブ構造体にエラーフラグを設定し、他モジュールから参照可能にする
\end{itemize}
